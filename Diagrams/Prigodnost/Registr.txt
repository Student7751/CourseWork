@startuml
boundary "Окно регистрации" as registry_window
boundary "Окно ошибки" as error_window
boundary "Окно авторизации" as entry_window

actor "Клиент" as user

Entity "База данных" as DB

control "Подтверждение данных" as validation_control

user -- registry_window

registry_window --> validation_control : Вводит данные и нажимает кнопку "Подтвердить"
validation_control --> entry_window: Успех
validation_control --> error_window: Ошибка ввода данных
validation_control --> DB: Успех


@enduml
