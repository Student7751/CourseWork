@startuml
boundary "Окно просмотра клиентов" as client_view_window
boundary "Главное окно" as main_window

actor "Администратор" as user


control "Назад" as back_control
control "Загрузка данных" as load_control
control "Ошибка" as error #red

user -- client_view_window

entity "База данных" as DB

client_view_window --> back_control: Нажата кнопка Назад
back_control --> main_window
client_view_window --> load_control
DB  <-- load_control: Запрос списка клиентов
DB --> client_view_window: Возврат данных
DB -left-> error: БД пуста
@enduml
